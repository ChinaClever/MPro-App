1、pdu_dev表：设备索引表
	id 		kｅｙ
    	dev_key;	设备关键字用来在数据库中区别不同的设备
    	cascade_addr;	设备级联地址 0 本机 
    	run_status;	运行状态 0：正常 1：告警 2：故障 3: 离线
    	dev_type;	设备型号
    	room_name;	机房名称
    	dev_position;	设备位置 
    	sw_version;	软件版本
    	ver_md;		版本校验码
    	dev_mode;	设备模式 	0：标准 1：级联 2：机柜双电源 3：RTU
    	slave_num;	副机数量		0－9
    	line_num;	相数量
    	loop_num;	回路数量
    	output_num;	输出位数量
    	dev_spec;	设备规格
    	rtu_addr;	rtu地址
    	mac;		硬件ｍａｃ
    	ip_v4;

    	qrcode;
